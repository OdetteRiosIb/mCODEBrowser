    <script>
    $(document).ready(function() {
        let globalNodeRegistry = {};

        // 1. Listen for local file selection via browser
        document.getElementById('fileInput').addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(evt) {
                try {
                    const parsedData = JSON.parse(evt.target.result);
                    processOntologyData(parsedData);
                } catch (err) {
                    alert("Parsing Error: The file is not valid JSON.");
                    console.error(err);
                }
            };
            reader.readAsText(file);
        });

        // Helper function to pull multilingual strings securely
        function extractAnnotation(field) {
            if (!field) return '';
            if (typeof field === 'object') {
                return field.en || field['en-us'] || field['undefined'] || Object.values(field)[0] || '';
            }
            return field;
        }

        // 2. Parse ontology arrays and balance labels vs structure keys
        function processOntologyData(vowlData) {
            globalNodeRegistry = {}; // reset registry on new upload
            
            let rawClasses = vowlData.class || [];
            let attributesList = vowlData.classAttribute || [];
            let propertiesList = vowlData.property || [];
            let propertyAttributes = vowlData.propertyAttribute || [];

            // First Pass: Extract labels and set structural parents
            rawClasses.forEach(classObj => {
                const uniqueId = classObj.id;
                const attr = attributesList.find(a => a.id === uniqueId) || {};
                
                if (classObj.type === "rdfs:Literal" || attr.type === "rdfs:Literal") return;

                let rdfsLabel = extractAnnotation(attr.label);
                let prefLabel = '';
                let altLabel = '';
                let skosDef = extractAnnotation(attr.description || attr.comment);

                // Scan nested annotations arrays for explicit skos identifiers
                if (attr.annotations) {
                    for (let key in attr.annotations) {
                        let cleanKey = key.toLowerCase();
                        let targetArray = attr.annotations[key];
                        let extractedVal = '';
                        
                        if (Array.isArray(targetArray) && targetArray.length > 0) {
                            extractedVal = targetArray[0].value || targetArray[0];
                        } else {
                            extractedVal = targetArray;
                        }

                        if (cleanKey.includes('preflabel')) prefLabel = extractedVal;
                        if (cleanKey.includes('altlabel')) altLabel = extractedVal;
                        if (cleanKey.includes('definition')) skosDef = extractedVal;
                    }
                }

                // Focus directly on choosing human labels over numbers
                let fallbackName = (classObj.iri || uniqueId).split('#').pop().split('/').pop().replace(/_/g, ' ');
                let displayName = prefLabel || rdfsLabel || fallbackName || uniqueId;

                // Prepend equivalence marker if the node is explicitly typed as equivalent
                if (classObj.type === "owl:equivalentClass") {
                    displayName = "≡ " + displayName;
                }

                // Configure tree nesting hierarchy directly via vowl array properties
                let parentNodeId = (attr.superClasses && attr.superClasses.length > 0) ? attr.superClasses[0] : '#';

                // Safely grab cross-referenced equivalent ID structures
                let equivalents = attr.equivalent || [];
                if (!Array.isArray(equivalents)) {
                    equivalents = [equivalents];
                }

                globalNodeRegistry[uniqueId] = {
                    id: uniqueId, 
                    text: displayName, 
                    parent: parentNodeId, 
                    meta: { 
                        type: classObj.type || 'Class', 
                        iri: classObj.iri || attr.iri || uniqueId, 
                        rdfsLabel: rdfsLabel || 'N/A',
                        prefLabel: prefLabel || 'N/A',
                        altLabel: altLabel || 'None',
                        comment: skosDef || 'No explicit description found.', 
                        superclasses: [], 
                        subclasses: [], 
                        slots: [],
                        equivalent: equivalents,
                        equivalentNames: []
                    }
                };
            });

            // Equivalence Sync Pass: Pull floating equivalent terms down into the same subtrees
            Object.keys(globalNodeRegistry).forEach(key => {
                let node = globalNodeRegistry[key];
                if (node.meta.equivalent && node.meta.equivalent.length > 0) {
                    node.meta.equivalent.forEach(eqId => {
                        let eqNode = globalNodeRegistry[eqId];
                        if (eqNode) {
                            // Extract text names without prefix markers for cleanly labeled mapping
                            let cleanName = eqNode.text.replace("≡ ", "");
                            if (!node.meta.equivalentNames.includes(cleanName)) {
                                node.meta.equivalentNames.push(cleanName);
                            }
                            
                            // Subtree nesting injection logic:
                            // If the twin equivalent node is isolated at root ('#'), copy over the valid parent path
                            if (eqNode.parent === '#' && node.parent !== '#') {
                                eqNode.parent = node.parent;
                            } else if (node.parent === '#' && eqNode.parent !== '#') {
                                node.parent = eqNode.parent;
                            }
                        }
                    });
                }
            });

            // Second Pass: Link inverse paths for sub/super text hierarchies
            Object.keys(globalNodeRegistry).forEach(key => {
                let node = globalNodeRegistry[key];
                if (node.parent !== '#' && globalNodeRegistry[node.parent]) {
                    let parentNode = globalNodeRegistry[node.parent];
                    node.meta.superclasses.push(parentNode.text);
                    parentNode.meta.subclasses.push(node.text);
                } else {
                    node.parent = '#'; // Safety floor baseline
                }
            });

            // Third Pass: Map domains to properties
            propertiesList.forEach(prop => {
                const propId = prop.id;
                const attr = propertyAttributes.find(a => a.id === propId) || {};
                
                let name = extractAnnotation(attr.label) || propId.split('#').pop();
                let comment = extractAnnotation(attr.comment || attr.description) || 'No explicit description.';
                let domainId = attr.domain;
                let rangeId = attr.range;
                let safeType = prop.type ? prop.type.toLowerCase() : "";

                if (safeType.includes("objectproperty") || safeType.includes("datatypeproperty")) {
                    if (domainId && globalNodeRegistry[domainId]) {
                        let targetRangeName = globalNodeRegistry[rangeId] ? globalNodeRegistry[rangeId].text : (rangeId || 'Literal Value');
                        let displayType = safeType.includes("objectproperty") ? 'Object Property' : 'Data Property';
                        
                        globalNodeRegistry[domainId].meta.slots.push({ 
                            name: name, type: displayType, range: targetRangeName, comment: comment 
                        });
                    }
                }
            });

            // Load and mount visualization layout engine
            renderTreeInterface();
        }

        function renderTreeInterface() {
            $('#protege-tree').jstree("destroy").jstree({
                'core': { 
                    'data': Object.values(globalNodeRegistry), 
                    'themes': { 'variant': 'large', 'icons': true, 'dots': true } 
                },
                'plugins': ["search", "wholerow"]
            });
        }

        // Search bar activation
        $('#protege-search').keyup(function () {
            $('#protege-tree').jstree(true).search($(this).val());
        });

        // 3. Populate Right-Side panels on click
        $('#protege-tree').on("changed.jstree", function (e, data) {
            if(!data.selected.length) return;
            
            const node = globalNodeRegistry[data.selected[0]];
            if (!node) return;
            
            const m = node.meta;
            let html = "";

            let supers = m.superclasses.length ? m.superclasses.map(s => `<span class="value-link">○ ${s}</span>`).join('<br/> ') : 'None';
            let subs = m.subclasses.length ? m.subclasses.map(s => `<span class="value-link">● ${s}</span>`).join('<br/> ') : 'None';
            let equivs = m.equivalentNames.length ? m.equivalentNames.map(e => `<span class="value-link" style="color: #6f42c1;">≡ ${e}</span>`).join('<br/> ') : 'None';
            let rows = "";
            
            if (m.slots.length) {
                m.slots.forEach(s => {
                    let pillColor = s.type === 'Data Property' ? '#1e7e34' : '#0056b3';
                    rows += `<tr>
                        <td>
                            <div class="slot-icon" style="background-color:${pillColor};"></div>
                            <strong>${s.name}</strong> 
                            <span class="prop-tag" style="color:${pillColor}; font-weight:bold;">[${s.type}]</span>
                        </td>
                        <td>${s.comment}</td>
                        <td><code>${s.range}</code></td>
                    </tr>`;
                });
            } else {
                rows = `<tr><td colspan="3" style="color:#888; font-style:italic; text-align:center;">No Object Properties or Data Properties mapped to this entity domain.</td></tr>`;
            }

            html = `
                <div class="class-title">Class: ${node.text}</div>
                
                <div class="yellow-card">
                    <div class="card-label">skos:definition / Annotation:</div>
                    <div class="card-value">${m.comment}</div>
                </div>

                <div class="yellow-card" style="background-color: #f4f6f8; border-color: #d1d5db;">
                    <div class="card-label" style="color: #4b5563;">Lexical Tracking Labels</div>
                    <div class="card-value" style="font-size: 0.9em; line-height: 1.6em;">
                        <strong>rdfs:label:</strong> <code>${m.rdfsLabel}</code><br/>
                        <strong>skos:prefLabel:</strong> <code>${m.prefLabel}</code><br/>
                        <strong>skos:altLabel:</strong> <span style="color:#555;">${m.altLabel}</span>
                    </div>
                </div>

                <div class="yellow-card" style="background-color: #f3f0fa; border-color: #d6cbe3;">
                    <div class="card-label" style="color: #6f42c1;">Equivalent Classes (owl:equivalentClass)</div>
                    <div class="card-value">${equivs}</div>
                </div>

                <div class="yellow-card">
                    <div class="card-label">Superclasses Hierarchy</div>
                    <div class="card-value">${supers}</div>
                </div>

                <div class="yellow-card">
                    <div class="card-label">Subclasses Hierarchy</div>
                    <div class="card-value">${subs}</div>
                </div>

                <div class="slot-section-title">Class Properties (Object & Data Properties)</div>
                <table class="slot-table">
                    <thead><tr><th>Property Name</th><th>Documentation</th><th>Range Target Value</th></tr></thead>
                    <tbody>${rows}</tbody>
                </table>
                <div style="margin-top:25px; font-size:0.8em; color:#666;">System IRI Location: <code>${m.iri}</code></div>`;
                
            $('#protege-inspector').html(html);
        });
    });
    </script>
